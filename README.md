## json-api-model

按 _dataM json 格式 将 api 返回数据格式等json 转成需要的 数据格式

``` js
// 数组包含对象时 采用 _dataM 中第一个数组元素
[
    {
        a: 0
    }
]
// 'false' 'FALSE' 转成 false
// 'true' 'TRUE' 转成 true
const __Model = require('../src/index');
let _dataM = new __Model({
  a: 0,
  b: '',
  c: new Date(),
  d: false,
  e: [],
  e2: [
    {
      a: 0
    }
  ],
  f: {},
  g: (t) => {
    return t
  }
});
let _r = null
// console.log(_dataM)
_r = _dataM.init({
  a: '3',
  b: 3,
  c: '2018-03-02T09:36:31.035Z',
  d: 'false',
  e: 'text',
  e2:
    {
      a: '1'
    }
  ,
  f: '{}',
  g: '_g_'
});
// {
//   a: 3,
//   b: '3',
//   c: 2018-03-02T09:36:31.035Z,
//   d: false,
//   e: [ 'text' ],
//   e2: [ { a: 1 } ],
//   f: {},
//   g: '_g_'
// }
console.log(_r);
```
